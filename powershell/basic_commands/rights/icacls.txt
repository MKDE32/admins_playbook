############### WATCH PERMISSIONS ###############
icacls 'C:\Pictures\'



############### GRANT PERMISSIONS ###############
icacls 'C:\Pictures\' /grant 'Everyone:(OI)(CI)(R)'
icacls .\ /grant Everyone:(OI)(CI)F /T


############### REMOVE PERMISSIONS ###############
icacls 'C:\Pictures\' /remove Everyone



############### REMOVE INHERITANCE ###############
icacls .\ /inheritance:d
